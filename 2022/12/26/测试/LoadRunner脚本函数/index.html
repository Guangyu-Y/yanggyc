  • lr_save_string(“A”,”B”); //将lr中定义的参数B保存为A。
  • lr_start_transaction(“transaction_name”); //事务开始
    • transaction_name:事务名
  • lr_end_transaction(“transaction_name”,status); //事务结束
    • transaction_name:事务名
    • status:事务的结束状态{LR_PASS(通过)、LR_FAIL(失败)、LR_AUTO（自动）、 LR_STOP(暂停)}
  • lr_set_transaction_status(Status);//事务结束前设置事务状态
  • long id = lr_start_transaction_instance（const char * transaction_name，long handle）;//标记事务实例的开始。事务实例是具有transaction_name的事务的发生。实例通过其句柄来识别，其将它们与同一事务的其他实例区分开。
    • transaction_name：事务的名称。
    • handle：父事务的句柄。值为0指示VuGen创建新的父事务。
  • lr_end_transaction_instance（id，LR_PASS）;//
  • lr_rendezvous(“集合点名”);//设置集合点
  • web_url
  • web_custom_request();//函数允许使用任何方法或主体创建自定义的HTTP请求，其使用范围更广。其中，Body中，如果是json类型的，引号需要使用 \ 进行转义。其他类型的，直接跟=后面即可。
  • web_image
  • web_add_cookie
  • web_add_header
  • web_reg_find(“Search=Body”,”Text=user_id”,”SaveCount=userId”,LAST);//验证响应是否符合期望，如通过搜索文本字符串
    • Search : 搜索字符串的搜索范围，如ALL、Body、Header
    • Text : 搜索的字符串
    • SaveCount : 匹配的数量，并存储在一个参数中
  • web_reg_save_param(“user_token”,”LB="user_token":"“,”RB="“,”Ord=1”,”Notfound=empty”,”Search=Body”,LAST);//
    • LB 参数的左边界
    • RB 参数的右边界
    • Ord 匹配的序号位置，默认为1
    • Notfound 当没有找到边界时的处理选项，选项有：Notfound=error - Notfound=empty
    • Search : 搜索字符串的搜索范围，如ALL、Body、Header
